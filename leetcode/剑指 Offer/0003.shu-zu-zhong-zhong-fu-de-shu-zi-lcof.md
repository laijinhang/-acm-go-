[剑指 Offer 03. 数组中重复的数字](https://leetcode-cn.com/problems/shu-zu-zhong-zhong-fu-de-shu-zi-lcof/)
```golang
func findRepeatNumber(nums []int) int {
    m := make(map[int]int)
    for i := 0;i < len(nums);i++ {
        if m[nums[i]] >= 1 {
            return nums[i]
        }
        m[nums[i]]++
    }
    return 0
}
```
