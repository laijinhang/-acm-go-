参考视频：[面试题56.I. 数组中数字出现的次数 二进制运算 (c语言c++笔试面试刷题)](https://www.bilibili.com/video/BV1Ck4y1r7ag?t=459)
时间复杂度：O(n)
空间复杂度：O(1)
func singleNumbers(nums []int) []int {
    x, y := 0, 0
    for i := 0;i < len(nums);i++ {
        x ^= nums[i]
    }
    for y = 0;x & (1 << y) == 0;y++ {}
    a, b := 0, 0
    for i := 0;i < len(nums);i++ {
        if nums[i] & (1 << y) == 0 {
            a ^= nums[i]
        } else {
            b ^= nums[i]
        }
    }
    return []int{a, b}
}
