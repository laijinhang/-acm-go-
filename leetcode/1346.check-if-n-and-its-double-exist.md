[检查整数及其两倍数是否存在](https://leetcode-cn.com/problems/check-if-n-and-its-double-exist/)

```golang
func checkIfExist(arr []int) bool {
    m := make(map[int]bool)
    for i := 0;i < len(arr);i++ {
        m[arr[i]] = true
    }
    for i := 0;i < len(arr);i++ {
       if m[arr[i] * 2] {
           return true
       } 
    }
    return false
}
```
